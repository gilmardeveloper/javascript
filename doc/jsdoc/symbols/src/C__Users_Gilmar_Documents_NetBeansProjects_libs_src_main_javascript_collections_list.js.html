 11 /**@class Lista
 12  * @description classe que representa uma lista de dados genérica */
 13 function Lista() {
 14     /**array de elementos*/
 15     this._valores = [];
 16     
 17     /*** 
 18      * @method add
 19      * @description adiciona um elemento a lista
 20      * @param   {Object} item
 21      * @returns {Number} index do elemento
 22      * */
 23     this.add = function(item) {
 24         var index = this._valores.push(item) - 1;
 25         return index;
 26     };
 27 
 28     /*** 
 29          * @method remove
 30          * @description remove um elemento da lista
 31          * @param   {Number} index 
 32          * @returns {Object} o elemento removido
 33          * */
 34     this.remove = function(index){
 35         return this._valores.splice(index, 1);
 36     };
 37 
 38     /*** 
 39          * @method get
 40          * @description recupera um elemento da lista
 41          * @param   {Number} index 
 42          * @returns {Object} o elemento da lista
 43          * */
 44     this.get = function(index) {
 45         return this._valores[index];
 46     };
 47 
 48     /*** 
 49          * @method getList
 50          * @description retorna uma lista de elementos
 51          * @returns {[Object]} array de elementos da lista
 52          * */
 53     this.getList = function() {
 54         return this._valores;
 55     };
 56 
 57     /*** 
 58          * @method getIndexOf
 59          * @description Retorna o índice de um elemento especifico da lista
 60          * @param   {Object} item 
 61          * @returns {Number} index de um elemento da lista
 62          * @requires Object.equals o método equals deve ser implementado no elemento 
 63          * */
 64     this.getIndexOf = function(item) {
 65         var index = -1;
 66         this._valores.find(function(element, pos){
 67             if (element.equals(item)) {
 68                 index = pos;
 69             }
 70         }, item);
 71 
 72         return index;
 73     };
 74     /*** 
 75          * @method getCount 
 76          * @description retorna o tamanho da lista
 77          * @returns {Number} tamanho da lista
 78          * */
 79     this.getCount = function() {
 80         return this._valores.length;
 81     };
 82 
 83 }
